bgc function: 

const element = document.querySelector('#block');

let backgroundColor = getComputedStyle(element, null).getPropertyValue('background-color');

let textColor;

// to get the color in rgb format 
let rgb = backgroundColor.replace(/^rgba?\(|\s+|\)$/g,'').split(',');

// get the luminance of the color
let luminance = 0.2126 * rgb[0] + 0.7152 * rgb[1] + 0.0722 * rgb[2];

if (luminance < 128) {
	textColor = '#FFFFFF';
}
else {
	textColor = '#000000';
}

element.style.color = textColor;
